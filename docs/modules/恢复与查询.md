## Run-scoped Event 查询

新增基于 run 范围的事件查询 API：

### API 端点

```
GET /api/task/:taskId/runs/:runId/events?after=<eventId>
GET /api/task/:taskId/runs/:runId/events?afterSequence=<sequence>
```

### 参数说明

- `after`: 基于事件 ID 的游标查询（向后兼容）
- `afterSequence`: 基于 run-scoped sequence 的增量查询（**推荐**）

### 响应格式

```typescript
interface EventQueryResponse {
  events: TaskEventEnvelope[];
  hasMore: boolean;
  // 🚨 STAGE0_SPEC §11.2 只定义 hasMore；nextSequence 不在契约内，如需游标请用最后一个 event 的 sequence
}
```

## 前端恢复策略

### 页面加载流程

```typescript
async function loadTaskPage(taskId: string) {
  // 1. 加载完整状态
  const taskDetail = await fetchTaskDetail(taskId);
  
  // 2. 渲染初始状态
  renderTask(taskDetail);
  
  // 3. 判断是否需要订阅实时更新
  const latestRun = taskDetail.latestRun;
  
  if (!latestRun) {
    return;
  }
  
  if (latestRun.status === 'queued' || latestRun.status === 'running') {
    // 订阅已有 run 的实时事件（轮询策略）
    await subscribeToRunEvents(taskId, latestRun.id, latestRun.lastEventSequence);
  }
}

/**
 * 订阅已有 run 的实时事件
 * 
 * 🚨 阶段 0-2 简化策略：使用轮询，不实现 SSE 订阅端点
 * 
 * 原因：
 * - POST /message 只能创建新 run，不能订阅已有 run
 * - 阶段 0-2 不实现独立的 SSE 订阅端点（简化架构）
 * - 使用轮询足够满足页面刷新恢复场景（低频操作）
 * 
 * 未来扩展（阶段 3+）：实现 GET /api/task/:taskId/runs/:runId/stream (SSE)
 */
async function subscribeToRunEvents(
  taskId: string,
  runId: string,
  lastSequence: number
) {
  let currentSequence = lastSequence;
  
  // 轮询间隔：1 秒
  const pollInterval = 1000;
  
  const poll = async () => {
    try {
      const response = await fetch(
        `/api/task/${taskId}/runs/${runId}/events?afterSequence=${currentSequence}&limit=100`
      );
      
      if (!response.ok) {
        // Run 已完成或出错，停止轮询
        if (response.status === 404 || response.status === 410) {
          return;
        }
        throw new Error(`Poll failed: ${response.status}`);
      }
      
      const { events } = await response.json();
      
      if (events.length > 0) {
        // 处理新事件
        events.forEach((envelope: TaskEventEnvelope) => {
          handleEvent(envelope);
          currentSequence = envelope.sequence;
        });
      }
      
      // 检查 run 是否完成
      const lastEvent = events[events.length - 1];
      if (lastEvent && (lastEvent.type === 'turn_complete' || lastEvent.type === 'error')) {
        return; // 停止轮询
      }
      
      // 继续轮询
      setTimeout(poll, pollInterval);
      
    } catch (error) {
      console.error('Poll error:', error);
      // 网络错误：等待 3 秒后重试
      setTimeout(poll, 3000);
    }
  };
  
  // 开始轮询
  poll();
}
```

### Fetch Streaming 断线恢复

**🚨 使用 fetch streaming，不是 EventSource**

**🚨 仅用于新建 run（POST /message）**

```typescript
class RunEventStream {
  private lastSequence = 0;
  private currentRunId: string | null = null;
  private abortController: AbortController | null = null;
  
  async connect(taskId: string, prompt: string) {
    this.abortController = new AbortController();
    
    try {
      const response = await fetch(`/api/task/${taskId}/message`, {
        method: 'POST',
        headers: { 'Content-Type': 'application/json' },
        body: JSON.stringify({ prompt }),
        credentials: 'include',
        signal: this.abortController.signal,
      });
      
      const reader = response.body!.getReader();
      const decoder = new TextDecoder();
      let buffer = '';
      
      while (true) {
        const { done, value } = await reader.read();
        if (done) break;
        
        buffer += decoder.decode(value, { stream: true });
        const lines = buffer.split('\n\n');
        buffer = lines.pop() || '';
        
        for (const line of lines) {
          if (line.startsWith('data: ')) {
            const envelope: TaskEventEnvelope = JSON.parse(line.slice(6));
            this.lastSequence = envelope.sequence;
            
            // 从 run_started 事件提取 runId
            if (envelope.type === 'run_started') {
              this.currentRunId = envelope.payload.runId;
            }
            
            this.handleEvent(envelope);
          }
        }
      }
    } catch (error) {
      if (error.name !== 'AbortError') {
        // 网络错误：补齐缺失事件
        if (this.currentRunId) {
          await this.backfillMissingEvents(taskId, this.currentRunId);
        }
        // 可选：延迟后重试
      }
    }
  }
  
  private async backfillMissingEvents(taskId: string, runId: string) {
    const response = await fetch(
      `/api/task/${taskId}/runs/${runId}/events?afterSequence=${this.lastSequence}`
    );
    
    const { events } = await response.json();
    events.forEach((event) => this.handleEvent(event));
  }
  
  disconnect() {
    if (this.abortController) {
      this.abortController.abort();
      this.abortController = null;
    }
  }
  
  private handleEvent(envelope: TaskEventEnvelope) {
    // 处理事件
  }
}
```
