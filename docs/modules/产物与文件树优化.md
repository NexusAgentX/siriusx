# Artifact 和文件树优化

> 按需加载，避免全量扫描

---

## 核心问题

Artifact 是用户主要关心的产物，不应每次页面刷新都扫描完整 workspace。

---

## 优化策略

### 1. Artifact Store

Worker 完成 run 后由 `ArtifactIndexer` 更新 artifact store。

```typescript
class ArtifactIndexer {
  async indexArtifacts(taskId: string, runId: string, workspace: Workspace) {
    // 扫描 workspace 中的产物文件
    const files = await workspace.listFiles({
      include: ['docs/**/*', 'src/**/*', '*.md'],
      exclude: ['.git', 'node_modules', '.next', 'dist'],
    });
    
    for (const file of files) {
      await taskStore.upsertArtifact({
        taskId,
        path: file.path,
        kind: this.detectKind(file),
        status: file.status, // 🚨 schema 枚举：'generated' | 'modified' | 'template' | 'deleted'
        workspace_revision: workspace.revision,
        object_key: file.objectKey,
        etag: file.etag,
        content_type: file.contentType,
        size_bytes: file.size,
      });
    }
  }
  
  private detectKind(file: File): string {
    const ext = path.extname(file.path);
    
    // 🚨 对齐 schema 枚举：document | file | code | image
    if (['.md', '.txt', '.pdf', '.docx'].includes(ext)) {
      return 'document';
    }
    if (['.ts', '.js', '.py', '.java', '.go', '.rs'].includes(ext)) {
      return 'code';
    }
    if (['.png', '.jpg', '.jpeg', '.gif', '.svg', '.webp'].includes(ext)) {
      return 'image';
    }
    
    return 'file';  // 默认
  }
}
```

## API 设计

### 获取 Artifact 列表

```
GET /api/task/:taskId/artifacts
```

**响应**：

```typescript
interface ArtifactListResponse {
  artifacts: TaskArtifact[];
  total: number;
}
```

### 懒加载文件树

```
GET /api/task/:taskId/files?path=docs&depth=2
```

**参数**：
- `path`: 起始路径（默认根目录）
- `depth`: 最大深度（默认 2）
- `limit`: 最大节点数（默认 100）

**响应**：

```typescript
interface FileTreeResponse {
  path: string;
  children: FileNode[];
  hasMore: boolean;
}

interface FileNode {
  name: string;
  path: string;
  type: 'file' | 'directory';
  size?: number;
  modified?: number;
  hasChildren?: boolean; // 目录是否有子节点
}
```

### Artifact 列表组件

```typescript
function ArtifactList({ taskId }: { taskId: string }) {
  const [artifacts, setArtifacts] = useState<TaskArtifact[]>([]);
  
  useEffect(() => {
    // 只加载 artifact store，不扫描 workspace
    fetch(`/api/task/${taskId}/artifacts`)
      .then(res => res.json())
      .then(data => setArtifacts(data.artifacts));
  }, [taskId]);
  
  return (
    <div>
      {artifacts.map(artifact => (
        <ArtifactItem key={artifact.path} artifact={artifact} />
      ))}
    </div>
  );
}
```

### 懒加载文件树

```typescript
function FileTree({ taskId, initialPath = '' }: Props) {
  const [nodes, setNodes] = useState<FileNode[]>([]);
  const [expandedPaths, setExpandedPaths] = useState<Set<string>>(new Set(['docs']));
  
  async function loadChildren(path: string) {
    const response = await fetch(
      `/api/task/${taskId}/files?path=${path}&depth=1`
    );
    const data = await response.json();
    return data.children;
  }
  
  async function toggleNode(path: string) {
    if (expandedPaths.has(path)) {
      // 折叠
      setExpandedPaths(prev => {
        const next = new Set(prev);
        next.delete(path);
        return next;
      });
    } else {
      // 展开：懒加载子节点
      const children = await loadChildren(path);
      setNodes(prev => [...prev, ...children]);
      setExpandedPaths(prev => new Set(prev).add(path));
    }
  }
  
  return (
    <Tree>
      {nodes.map(node => (
        <TreeNode 
          key={node.path} 
          node={node}
          expanded={expandedPaths.has(node.path)}
          onToggle={() => toggleNode(node.path)}
        />
      ))}
    </Tree>
  );
}
```

## 后端实现

### 文件树查询

```typescript
class WorkspaceFileService {
  async listFiles(
    taskId: string,
    options: {
      path?: string;
      depth?: number;
      limit?: number;
    }
  ): Promise<FileTreeResponse> {
    const { path = '', depth = 2, limit = 100 } = options;
    
    // 从 workspace 读取文件树
    const workspace = await this.workspaceProvider.getWorkspace(taskId);
    const files = await workspace.listFiles({
      path,
      maxDepth: Math.min(depth, 5),
      maxNodes: Math.min(limit, 1000),
      exclude: DEFAULT_EXCLUDE_PATTERNS,
    });
    
    // 转换为前端需要的格式
    const nodes = files.map(file => ({
      name: path.basename(file.path),
      path: file.path,
      type: file.isDirectory ? 'directory' : 'file',
      size: file.size,
      modified: file.modified,
      hasChildren: file.isDirectory && file.childCount > 0,
    }));
    
    return {
      path,
      children: nodes,
      hasMore: files.length >= limit,
    };
  }
}
```

### 增量更新

```typescript
// 🚨 明确状态映射，防止写入非法值
function mapFileStatus(status: string): ArtifactStatus {
  // workspace 可能返回旧的 'created'
  if (status === 'created') return 'generated';
  if (status === 'modified') return 'modified';
  if (status === 'deleted') return 'deleted';
  if (status === 'template') return 'template';
  
  // 默认
  return 'generated';
}

// Run 完成后，只更新变化的 artifacts
async function updateArtifactsIncremental(taskId: string, runId: string) {
  const changedFiles = await workspace.getChangedFiles(runId);
  
  for (const file of changedFiles) {
    await taskStore.upsertArtifact({
      taskId,
      path: file.path,
      status: mapFileStatus(file.status), // ✅ 映射到 schema 枚举
      // ... 其他字段
    });
  }
  
  // 清除缓存
  invalidateArtifactCache(taskId);
}
```
