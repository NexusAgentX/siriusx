describe('Control Worker Crash Recovery', () => {
  it('should recover run after control worker crashes', async () => {
    // 1. 创建 Task 和 Run
    const { taskId, runId } = await createTaskAndRun();
    
    // 2. 模拟 Worker 开始执行
    const worker = new TaskWorker({ ... });

    // 3. 模拟 Control Worker 崩溃（进程退出）

describe('Concurrent Message Serial Strategy', () => {
  it('should return 409 RUN_CONFLICT when task has active run', async () => {
    const task = await createTask({ title: 'Test' });
    
    // 第一条消息开始执行
    const run1Promise = sendMessage(task.id, 'Message 1');
    
    // 等待 run1 开始执行（状态变为 running）
    await waitForRunStatus(task.id, 'running');
    
    // 第二条消息应该返回 409
    await expect(
      sendMessage(task.id, 'Message 2')
    ).rejects.toMatchObject({
      status: 409,
      code: 'RUN_CONFLICT',
      message: expect.stringContaining('active run'),
    });
    
    // 等待第一条消息完成
    await run1Promise;
    
    // 现在可以发送第二条消息
    const run2 = await sendMessage(task.id, 'Message 2');
    expect(run2.status).toBe('queued');
  });
  
  it('should allow new message after run completes', async () => {
    const task = await createTask({ title: 'Test' });
    
    // 第一条消息
    const run1 = await sendMessage(task.id, 'Message 1');
    await waitForRunComplete(task.id, run1.id);
    
    // 第一条完成后，可以发送第二条
    const run2 = await sendMessage(task.id, 'Message 2');
    expect(run2.status).toBe('queued');
    await waitForRunComplete(task.id, run2.id);
  });
});
